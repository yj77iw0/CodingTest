```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>
#include <string.h> // strlen 함수를 쓰기 위한 라이브러리.

int* solution(const char* strlist[], size_t strlist_len) {
    
    int* answer = (int*)malloc(sizeof(int)*strlist_len); // 배열의 길이만큼 동적메모리 선언.
    
    for(int i = 0; i < strlist_len; i++){
        answer[i] = strlen(strlist[i]); // 각각의 배열과 같은 자리에 배열의 길이 추가.
    }
    
    return answer;
}
```
